    let currentQuestionText = null;
    let currentCorrectAnswers = null;
    let hoverEnabled = true;
    let hoverTimer = null;
    let currentHoverElement = null;
    let isRightClickPressed = false; // Флаг для отслеживания нажатия правой кнопки мыши

// Функция для нормализации текста
function normalizeText(text) {
    if (!text) return '';

    return text.trim().toLowerCase()
        .replace(/["""]/g, '') // Удаляем все виды кавычек
        .replace(/[.,!?;:()]/g, '') // Удаляем знаки препинания
        .replace(/\s+/g, ' ') // Заменяем множественные пробелы одним
        .trim();
}

    // Функция для точного сравнения текстов
    function textsMatch(optionText, correctText) {
        const normalizedOption = normalizeText(optionText);
        const normalizedCorrect = normalizeText(correctText);

        if (!normalizedOption || !normalizedCorrect) return false;

        return normalizedOption === normalizedCorrect;
    }

// Функция для поиска правильных ответов для текущего вопроса
function findCorrectAnswersForQuestion(questionText) {
    const normalizedQuestion = normalizeText(questionText);

    // Сначала ищем точное совпадение
    for (const [question, answers] of Object.entries(correctAnswerTexts)) {
        const normalizedDbQuestion = normalizeText(question);

        if (normalizedQuestion === normalizedDbQuestion) {
            console.log('✓ Найдено точное совпадение вопроса');
            return answers;
        }
    }

    // Если точного нет, ищем по началу вопроса (первые 30 символов)
    const questionStart = normalizedQuestion.substring(0, 30);
    for (const [question, answers] of Object.entries(correctAnswerTexts)) {
        const normalizedDbQuestion = normalizeText(question);
        const dbQuestionStart = normalizedDbQuestion.substring(0, 30);

        if (questionStart === dbQuestionStart) {
            console.log('✓ Найдено совпадение по началу вопроса');
            return answers;
        }
    }

    // Убираем слишком "размытый" поиск
    return null;
}

    // Функция для получения текущего вопроса
    function getCurrentQuestion() {
        let questionElement = document.querySelector('.question');
        if (!questionElement) {
            questionElement = document.querySelector('.task_title');
        }

        if (!questionElement) {
            return null;
        }

        return questionElement.textContent.trim();
    }

    // Функция для обновления кэша правильных ответов
    function updateCorrectAnswersCache() {
        const questionText = getCurrentQuestion();

        if (!questionText) {
            currentQuestionText = null;
            currentCorrectAnswers = null;
            return;
        }

        // Если вопрос изменился
        if (questionText !== currentQuestionText) {
            console.log(`📝 Текущий вопрос: "${questionText}"`);
            currentQuestionText = questionText;
            currentCorrectAnswers = findCorrectAnswersForQuestion(questionText);

            if (currentCorrectAnswers) {
                console.log(`✅ Найдены правильные ответы (${currentCorrectAnswers.length}):`, currentCorrectAnswers);
            } else {
                console.log('❌ Правильные ответы не найдены');
            }
        }
    }

    // Функция для проверки, является ли вариант ответа правильным
    function isOptionCorrect(optionElement) {
        if (!currentCorrectAnswers || !hoverEnabled) return false;

        const rightSpan = optionElement.querySelector('.right');
        const optionText = rightSpan ? rightSpan.textContent : optionElement.textContent;

        return currentCorrectAnswers.some(correctAnswer =>
            textsMatch(optionText, correctAnswer)
        );
    }

    // Функция для выбора варианта ответа
    function selectOption(optionElement) {
        if (!hoverEnabled) return;

        const input = optionElement.querySelector('input[type="radio"], input[type="checkbox"]');
        if (!input) return;

        if (!input.checked) {
            input.checked = true;
            input.dispatchEvent(new Event('change', { bubbles: true }));
            input.dispatchEvent(new Event('click', { bubbles: true }));

            console.log(`✅ Автоматически выбран: "${optionElement.textContent.trim()}"`);
        }
    }

    // Обработчик наведения мыши
    function handleMouseOver(event) {
        // Если нажата правая кнопка мыши - игнорируем наведение
        if (isRightClickPressed) {
            console.log('🚫 Правая кнопка нажата - автовыбор отключен');
            return;
        }

        updateCorrectAnswersCache();

        let target = event.target;

        // Ищем родительский элемент с классом option
        while (target && !target.classList?.contains('option')) {
            target = target.parentElement;
            if (!target || target === document.body) return;
        }

        if (!target || !target.classList.contains('option')) return;

        // Проверяем, правильный ли это ответ
        if (isOptionCorrect(target)) {
            currentHoverElement = target;

            // Запускаем таймер с задержкой 500ms
            hoverTimer = setTimeout(() => {
                if (currentHoverElement === target) {
                    selectOption(target);
                    currentHoverElement = null;
                }
            }, 500); // 0.5 секунды задержки
        }
    }

    // Обработчик ухода мыши с элемента
    function handleMouseOut(event) {
        let target = event.target;

        // Ищем родительский элемент с классом option
        while (target && !target.classList?.contains('option')) {
            target = target.parentElement;
            if (!target || target === document.body) return;
        }

        if (!target || !target.classList.contains('option')) return;

        // Если ушли с элемента, который был наведен, отменяем таймер
        if (target === currentHoverElement) {
            clearTimeout(hoverTimer);
            hoverTimer = null;
            currentHoverElement = null;
        }
    }

    // Обработчик нажатия кнопки мыши
    function handleMouseDown(event) {
        // Проверяем, нажата ли правая кнопка (button 2)
        if (event.button === 2) {
            isRightClickPressed = true;
            console.log('🔒 Правая кнопка нажата - автовыбор временно заблокирован');
            // Отменяем любой текущий таймер
            if (hoverTimer) {
                clearTimeout(hoverTimer);
                hoverTimer = null;
                currentHoverElement = null;
            }
        }
    }

    // Обработчик отпускания кнопки мыши
    function handleMouseUp(event) {
        // Проверяем, отпущена ли правая кнопка
        if (event.button === 2) {
            isRightClickPressed = false;
            console.log('🔓 Правая кнопка отпущена - автовыбор снова активен');
        }
    }

    // Обработчик контекстного меню (предотвращаем появление стандартного меню)
    function handleContextMenu(event) {
        // Не блокируем полностью, просто отмечаем что правая кнопка была нажата
        // Это предотвратит случайный автовыбор при вызове контекстного меню
        console.log('📋 Вызвано контекстное меню');
    }

    // Обработчик клавиш для отладки
    function handleKeyPress(event) {
        // Ctrl+Shift+D для вывода отладочной информации
        if (event.ctrlKey && event.shiftKey && event.key === 'D') {
            console.log('=== ОТЛАДОЧНАЯ ИНФОРМАЦИЯ ===');
            console.log('Текущий вопрос:', currentQuestionText);
            console.log('Правильные ответы:', currentCorrectAnswers);
            console.log('Автовыбор включен:', hoverEnabled);
            console.log('Правая кнопка нажата:', isRightClickPressed);
        }

        // Ctrl+Shift+T для переключения режима
        if (event.ctrlKey && event.shiftKey && event.key === 'T') {
            hoverEnabled = !hoverEnabled;
            console.log(hoverEnabled ? '✅ Автовыбор включен' : '❌ Автовыбор выключен');
        }
    }

    // Инициализация
    function init() {
        // Добавляем обработчики мыши
        document.addEventListener('mouseover', handleMouseOver);
        document.addEventListener('mouseout', handleMouseOut);
        document.addEventListener('mousedown', handleMouseDown);
        document.addEventListener('mouseup', handleMouseUp);
        document.addEventListener('contextmenu', handleContextMenu);

        // Добавляем обработчик клавиш для отладки
        document.addEventListener('keydown', handleKeyPress);

        // Обновляем кэш при клике (для перехода к следующему вопросу)
        document.addEventListener('click', updateCorrectAnswersCache);

        console.log('🚀 СКРИПТ АВТОВЫБОРА ЗАПУЩЕН! Exam');
        console.log('🎯 Наведите курсор на правильный ответ для автоматического выбора');
        console.log('⏱️  Автовыбор произойдет через 0.5 секунды после наведения');
        console.log('🚫 При нажатой правой кнопке мыши автовыбор НЕ срабатывает');
        console.log('🔄 Нажмите Ctrl+Shift+T для переключения режима автовыбора');
        console.log('📊 Нажмите Ctrl+Shift+D для отладочной информации');
    }

    // Запуск при загрузке страницы
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', init);
    } else {
        init();
    }

})();
